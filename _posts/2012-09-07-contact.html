def launch(port):
    application.listen(port)
    logger.info('Server started on %s' % port)
    tornado.ioloop.IOLoop.instance().start()


def start_random_warn():
    logger.info('Fake warning enabled!')
    from threading import Thread
    import utils.fake_warning
    td = Thread(target=utils.fake_warning.run)
    td.daemon = True
    td.start()


def pre_init():
    from db import connection
    if not connection.Config.get('Algorithm'):
        connection.Config.set('Algorithm', Algorithm)
    server.Application.startup()


if __name__ == '__main__':
    if 'fake_warning' in sys.argv:
        start_random_warn()
    port = options.web_server_port
    argv = sys.argv
    if len(argv) > 1:
        p = argv[1]
        if p.isdigit():
            port = int(p)
    pre_init()
    launch(port)
